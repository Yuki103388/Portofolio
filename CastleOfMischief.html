                                                using UnityEngine;
                                                using UnityEngine.AI;

                                                public class ArcherController : MonoBehaviour
                                                {
                                                    private NavMeshAgent agent;
                                                    private float lastAttackTime;
                                                    private Transform player;
                                                    private Animator animator;
                                                    private AudioSource arrowSound;

                                                    public StatsData enemyStats;
                                                    public float attackDelay = 4f;
                                                    public float attackRange = 30f;
                                                    public GameObject arrowPrefab;

                                                    void Start()
                                                    {
                                                        arrowSound = GetComponent();
                                                        agent = GetComponent();
                                                        animator = GetComponent();
                                                        lastAttackTime = -attackDelay;

                                                        player = GameObject.FindWithTag("Player")?.transform;
                                                        if (player == null) Debug.LogWarning("Player not found!");
                                                    }

                                                    void Update()
                                                    {
                                                        if (player == null) return;

                                                        float distanceToPlayer = Vector3.Distance(transform.position, player.position);
                                                        if (distanceToPlayer <= attackRange && Time.time - lastAttackTime >= attackDelay)
                                                        {
                                                            lastAttackTime = Time.time;
                                                            ShootArrow();
                                                        }
                                                        else
                                                        {
                                                            agent.destination = player.position;
                                                        }

                                                        animator.SetBool("Walk", agent.velocity.magnitude > 0.01f);
                                                    }

                                                    public void TakeDamage(int amount)
                                                    {
                                                        enemyStats.currentHealth -= amount;
                                                        if (enemyStats.currentHealth <= 0) Die();
                                                    }

                                                    private void ShootArrow()
                                                    {
                                                        if (arrowPrefab == null) return;

                                                        animator.SetTrigger("Attack");
                                                        Invoke("SpawnArrow", 0.5f);
                                                    }

                                                    private void SpawnArrow()
                                                    {
                                                        Vector3 spawnPos = transform.position + Vector3.up * 1.5f + transform.forward;
                                                        GameObject arrow = Instantiate(arrowPrefab, spawnPos, Quaternion.identity);
                                                        arrow.GetComponent()?.SetDirection((player.position - spawnPos).normalized);
                                                    }

                                                    private void Die()
                                                    {
                                                        Destroy(gameObject);
                                                    }
                                                }
